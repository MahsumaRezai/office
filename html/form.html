window.onload = function() { // دریافت اطلاعات از localStorage const userData = JSON.parse(localStorage.getItem('userData')); if (userData) { // نمایش داده‌ها در جدول const table = document.getElementById('dataTable'); userData.forEach(data => { const newRow = ` ${data.firstName} ${data.lastName} ${data.phoneNumber} `; table.innerHTML += newRow; }); } else { alert('هیچ داده‌ای برای نمایش وجود ندارد'); } };
